Rewrite ContentView.swift completely. The new routing logic:

**State machine rules:**

1. `auth.isLoading` → loading screen (unchanged)

2. `auth.isAuthenticated` AND `AmiirOnboardingCoordinator.hasCompletedOnboarding(userId:)` = true:
   → returning user → MainTabView(initialTab: 1)

3. `auth.isAuthenticated` AND not completed onboarding:
   → post-auth: trigger flush based on which coordinator is active, then MainTabView

4. NOT authenticated:
   → show onboarding flow (unauthenticated traversal):
   - If `amiirCoordinator.shouldSwitchToJoinerFlow` = true → show MemberOnboardingFlowView
   - If pendingInviteCode != nil → set memberCoordinator.inviteCodeInput and show MemberOnboardingFlowView
   - Otherwise → show AmiirOnboardingFlowView (Landing Sanctuary is the entry for everyone)

**Post-auth flush logic:**
After `auth.isAuthenticated` becomes true:
- If `amiirCoordinator.isComplete` is false AND not a returning user → call `amiirCoordinator.flushToSupabase(userId:)` in a Task
- If `memberCoordinator` exists AND `memberCoordinator.isComplete` is false → call `memberCoordinator.flushToSupabase(userId:)`
- After flush: the coordinator sets isComplete = true via completeOnboarding → ContentView re-evaluates and routes to MainTabView

**New state:**
```swift
@State private var amiirCoordinator = AmiirOnboardingCoordinator()
@State private var memberCoordinator = MemberOnboardingCoordinator()
@State private var isFlushingToSupabase = false
```

Note: memberCoordinator is now always initialized (no Optional) since MemberOnboardingCoordinator.init() takes no arguments.

```swift
import SwiftUI
import Supabase

struct ContentView: View {
    @Environment(AuthManager.self) private var auth
    @Environment(ThemeManager.self) private var themeManager
    @Environment(\.pendingInviteCode) private var pendingInviteCode

    @State private var amiirCoordinator = AmiirOnboardingCoordinator()
    @State private var memberCoordinator = MemberOnboardingCoordinator()
    @State private var isFlushingToSupabase = false

    /// True when the active flow is Joiner (member)
    private var isJoinerFlow: Bool {
        amiirCoordinator.shouldSwitchToJoinerFlow || pendingInviteCode != nil
    }

    var body: some View {
        Group {
            if auth.isLoading {
                loadingScreen
            } else if auth.isAuthenticated {
                authenticatedRouting
            } else {
                unauthenticatedOnboarding
            }
        }
        .preferredColorScheme(themeManager.colorScheme)
        .onChange(of: auth.isAuthenticated) { _, isAuthenticated in
            guard isAuthenticated, let userId = auth.session?.user.id else { return }
            handlePostAuth(userId: userId)
        }
        .onChange(of: pendingInviteCode) { _, code in
            if let code, !code.isEmpty {
                memberCoordinator.inviteCodeInput = code
                if !amiirCoordinator.shouldSwitchToJoinerFlow {
                    amiirCoordinator.shouldSwitchToJoinerFlow = true
                }
            }
        }
        .onAppear {
            // Propagate pending invite code on launch
            if let code = pendingInviteCode, !code.isEmpty {
                memberCoordinator.inviteCodeInput = code
                amiirCoordinator.shouldSwitchToJoinerFlow = true
            }
        }
    }

    // MARK: - Authenticated routing

    @ViewBuilder
    private var authenticatedRouting: some View {
        guard let userId = auth.session?.user.id else {
            loadingScreen
            return
        }
        if AmiirOnboardingCoordinator.hasCompletedOnboarding(userId: userId) {
            // Returning user
            MainTabView(initialTab: 1)
        } else if isFlushingToSupabase {
            // Flushing in progress — show loading
            loadingScreen
        } else if amiirCoordinator.isComplete || memberCoordinator.isComplete {
            // Flush done — route to app
            let tab = memberCoordinator.isComplete ? 1 : 0
            MainTabView(initialTab: tab)
        } else {
            // Auth just happened — flush will start via onChange
            loadingScreen
        }
    }

    // MARK: - Unauthenticated onboarding

    @ViewBuilder
    private var unauthenticatedOnboarding: some View {
        if isJoinerFlow {
            MemberOnboardingFlowView()
                .environment(memberCoordinator)
                .environment(auth)
        } else {
            AmiirOnboardingFlowView()
                .environment(amiirCoordinator)
                .environment(auth)
        }
    }

    // MARK: - Post-auth flush

    private func handlePostAuth(userId: UUID) {
        guard !AmiirOnboardingCoordinator.hasCompletedOnboarding(userId: userId) else { return }
        isFlushingToSupabase = true
        Task {
            if isJoinerFlow {
                await memberCoordinator.flushToSupabase(userId: userId)
            } else {
                await amiirCoordinator.flushToSupabase(userId: userId)
            }
            isFlushingToSupabase = false
        }
    }

    // MARK: - Loading screen

    private var loadingScreen: some View {
        ZStack {
            Color(hex: "1A2E1E").ignoresSafeArea()
            ProgressView()
                .tint(Color(hex: "D4A240"))
                .scaleEffect(1.4)
        }
    }
}
```

}
