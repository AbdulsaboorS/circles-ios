This file defines a Codable struct and static save/load/clear helpers used by BOTH coordinators to persist onboarding state before auth succeeds.

```swift
import Foundation

/// Persists all pre-auth onboarding selections to UserDefaults.
/// Saved after AI Generation step (Step 5 in both flows) when all selections are complete.
/// Cleared after auth succeeds and Supabase flush completes.
/// On auth failure: state is NOT cleared — user can retry.
struct OnboardingPendingState: Codable {

    // MARK: - Flow discriminator
    /// "amir" or "member"
    var flowType: String = "amir"

    // MARK: - Shared fields (both flows)
    var preferredName: String = ""
    var cityName: String = ""
    var cityTimezone: String = ""
    var cityLatitude: Double = 0
    var cityLongitude: Double = 0
    var selectedPersonalHabits: [String] = []

    // MARK: - Amir-specific
    var circleName: String = ""
    var genderSetting: String = "mixed"
    var selectedCoreHabits: [String] = []  // shared accountable habits, max 3

    // MARK: - Member-specific
    var inviteCode: String = ""
    var selectedCircleHabits: [String] = []  // habits picked from circle's core list, min 1

    // MARK: - Persistence
    private static let userDefaultsKey = "pending_onboarding_state"

    static func save(_ state: OnboardingPendingState) {
        guard let data = try? JSONEncoder().encode(state) else { return }
        UserDefaults.standard.set(data, forKey: userDefaultsKey)
    }

    static func load() -> OnboardingPendingState? {
        guard let data = UserDefaults.standard.data(forKey: userDefaultsKey),
              let state = try? JSONDecoder().decode(OnboardingPendingState.self, from: data)
        else { return nil }
        return state
    }

    static func clear() {
        UserDefaults.standard.removeObject(forKey: userDefaultsKey)
    }

    static func hasPendingState() -> Bool {
        UserDefaults.standard.data(forKey: userDefaultsKey) != nil
    }
}
```

Create Circles/Onboarding/OnboardingTransitionView.swift.

This is a full-screen view displayed between certain onboarding steps. It shows a centered Islamic quote, auto-advances after 1.5 seconds, and can be tapped/tapped anywhere to skip immediately.

```swift
import SwiftUI

// MARK: - Midnight Sanctuary tokens
private extension Color {
    static let msBackground  = Color(hex: "1A2E1E")
    static let msGold        = Color(hex: "D4A240")
    static let msTextPrimary = Color(hex: "F0EAD6")
    static let msTextMuted   = Color(hex: "8FAF94")
}

/// Full-screen Islamic narrative transition shown between onboarding steps.
/// Auto-advances after 1.5 seconds. Tap anywhere to skip.
struct OnboardingTransitionView: View {
    let quote: String          // Main quote text
    let attribution: String?   // Optional source (e.g. "— Hadith")
    let onSkip: () -> Void     // Called on auto-advance or tap

    @State private var opacity: Double = 0

    var body: some View {
        ZStack {
            Color.msBackground.ignoresSafeArea()

            VStack(spacing: 20) {
                // Decorative crescent/star
                Image(systemName: "moon.stars.fill")
                    .font(.system(size: 36))
                    .foregroundStyle(Color.msGold.opacity(0.6))

                Text(quote)
                    .font(.system(size: 20, weight: .medium, design: .serif))
                    .foregroundStyle(Color.msTextPrimary)
                    .multilineTextAlignment(.center)
                    .lineSpacing(6)
                    .padding(.horizontal, 36)

                if let attribution {
                    Text(attribution)
                        .font(.system(size: 13, weight: .regular, design: .default))
                        .foregroundStyle(Color.msTextMuted)
                        .multilineTextAlignment(.center)
                }
            }
            .opacity(opacity)
            .animation(.easeIn(duration: 0.4), value: opacity)
        }
        .contentShape(Rectangle())
        .onTapGesture { onSkip() }
        .onAppear {
            opacity = 1
            // Auto-advance after 1.5 seconds
            DispatchQueue.main.asyncAfter(deadline: .now() + 1.5) {
                onSkip()
            }
        }
        .navigationBarBackButtonHidden()
    }
}
```
