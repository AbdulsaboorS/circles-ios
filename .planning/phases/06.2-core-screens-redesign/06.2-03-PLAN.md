**D-20 target layout:**
- AppCard as the outermost container (replaces `.background(Color.white.opacity(0.06)).clipShape(RoundedRectangle(cornerRadius:14))`)
- Full-width photo fills the card top area
- Caption + user name overlaid at the bottom of the photo with a dark gradient (LinearGradient from .clear at top to .black.opacity(0.6) at bottom)

**New body structure:**
```swift
var body: some View {
    AppCard {
        VStack(spacing: 0) {
            // Photo with overlays
            ZStack(alignment: .bottom) {
                // Photo
                AsyncImage(url: URL(string: item.photoUrl)) { image in
                    image.resizable().scaledToFill()
                } placeholder: {
                    Color.white.opacity(0.05)
                }
                .frame(maxWidth: .infinity)
                .frame(height: 280)
                .clipped()
                .blur(radius: isLocked ? 20 : 0)

                // Lock overlay (unchanged when isLocked)
                if isLocked {
                    VStack(spacing: 8) {
                        Image(systemName: "lock.fill")
                            .font(.system(size: 28))
                            .foregroundStyle(.white)
                        Text("Post your Moment to see theirs")
                            .font(.appSubheadline)
                            .foregroundStyle(.white)
                            .multilineTextAlignment(.center)
                    }
                    .padding()
                    .background(.ultraThinMaterial, in: RoundedRectangle(cornerRadius: 12))
                }

                // Caption + username gradient overlay (D-20) — only when not locked
                if !isLocked {
                    VStack(alignment: .leading, spacing: 2) {
                        Text(item.userName)
                            .font(.appCaptionMedium)
                            .foregroundStyle(.white)
                        if let caption = item.caption, !caption.isEmpty {
                            Text(caption)
                                .font(.appCaption)
                                .foregroundStyle(.white.opacity(0.85))
                                .lineLimit(2)
                        }
                    }
                    .frame(maxWidth: .infinity, alignment: .leading)
                    .padding(10)
                    .background(
                        LinearGradient(
                            colors: [.clear, .black.opacity(0.6)],
                            startPoint: .top,
                            endPoint: .bottom
                        )
                    )
                }
            }
            // On-time star badge (D-20: top-right corner of photo)
            // Placed as .overlay(alignment: .topTrailing) on the ZStack above

            // Reaction bar below photo
            ReactionBar(
                itemId: item.id, itemType: "moment",
                currentUserId: currentUserId, viewModel: viewModel
            )
            .padding(.horizontal, 12)
            .padding(.vertical, 10)
        }
    }
    .overlay(alignment: .topTrailing) {
        // On-time star + relative timestamp header (top-right)
        if item.isOnTime {
            HStack(spacing: 4) {
                Image(systemName: "star.fill")
                    .font(.system(size: 11))
                    .foregroundStyle(Color.accent)
                Text(relativeTimestamp(item.postedAt))
                    .font(.appCaption)
                    .foregroundStyle(Color.textSecondary)
            }
            .padding(.horizontal, 10)
            .padding(.vertical, 6)
            .background(.ultraThinMaterial, in: Capsule())
            .padding(10)
        } else {
            Text(relativeTimestamp(item.postedAt))
                .font(.appCaption)
                .foregroundStyle(Color.textSecondary)
                .padding(.horizontal, 10)
                .padding(.vertical, 6)
                .background(.ultraThinMaterial, in: Capsule())
                .padding(10)
        }
    }
}
```

Remove the separate header row (HStack with name + star + timestamp at top) — this information now lives in the photo overlay and the .overlay badge.

Keep `relativeTimestamp(_ iso: String) -> String` private method unchanged.

Remove all `Color(hex:)` calls. The `Color(hex: "E8834B")` for the star becomes `Color.accent`.
