```swift
import Foundation

// MARK: - Associated value types for each feed item variant

struct MomentFeedItem: Identifiable, Sendable {
    let id: UUID          // circle_moments.id
    let circleId: UUID
    let userId: UUID
    let userName: String  // display name — fetched alongside (see FeedService note)
    let photoUrl: String
    let caption: String?
    let postedAt: String  // ISO8601 string, per project date-as-string convention
    let isOnTime: Bool
}

struct HabitCheckinFeedItem: Identifiable, Sendable {
    let id: UUID          // activity_feed.id
    let circleId: UUID
    let userId: UUID
    let userName: String
    let habitName: String
    let checkedAt: String // ISO8601 string (activity_feed.created_at)
}

struct StreakMilestoneFeedItem: Identifiable, Sendable {
    let id: UUID          // activity_feed.id
    let circleId: UUID
    let userId: UUID
    let userName: String
    let habitName: String
    let streakDays: Int
    let achievedAt: String // ISO8601 string (activity_feed.created_at)
}

// MARK: - Unified feed item enum

enum FeedItem: Identifiable, Sendable {
    case moment(MomentFeedItem)
    case habitCheckin(HabitCheckinFeedItem)
    case streakMilestone(StreakMilestoneFeedItem)

    var id: UUID {
        switch self {
        case .moment(let item): return item.id
        case .habitCheckin(let item): return item.id
        case .streakMilestone(let item): return item.id
        }
    }

    /// Timestamp string for reverse-chronological sorting
    var sortTimestamp: String {
        switch self {
        case .moment(let item): return item.postedAt
        case .habitCheckin(let item): return item.checkedAt
        case .streakMilestone(let item): return item.achievedAt
        }
    }
}
```

Notes:
- `userName` is resolved by FeedService at fetch time — the view should never need to look up names separately.
- `Codable` is NOT needed on these types because they are assembled by FeedService from raw DB rows, not decoded directly from JSON as-is.
- The `activity_feed` table stores both habit check-ins and streak milestones. FeedService distinguishes them via an `event_type` column (values: `"habit_checkin"` and `"streak_milestone"`).

**Public interface:**

```swift
@Observable @MainActor
final class FeedService {
    static let shared = FeedService()
    private init() {}
    private var client: SupabaseClient { SupabaseService.shared.client }

    // MARK: - Paginated feed fetch

    /// Returns up to `pageSize` FeedItems for `circleId`, newest first.
    /// `page` is 0-indexed. Merges activity_feed rows and circle_moments rows,
    /// sorts by timestamp descending, then slices the page window.
    func fetchFeedPage(circleId: UUID, page: Int, pageSize: Int = 20) async throws -> [FeedItem]

    // MARK: - Reactions

    /// Fetch all FeedReactions for a batch of item IDs.
    func fetchReactions(itemIds: [UUID]) async throws -> [FeedReaction]

    /// Toggle a reaction: if current user already has this emoji on this item, delete it.
    /// Otherwise, delete any existing reaction from this user on this item, then insert the new one.
    func toggleReaction(itemId: UUID, itemType: String, userId: UUID, emoji: String) async throws
}
```

**Implementation notes for `fetchFeedPage`:**

1. Fetch all `activity_feed` rows for `circleId` ordered by `created_at DESC` — no server-side pagination here (max ~200 rows per circle, manageable). Use `.eq("circle_id", value: circleId.uuidString)`.
2. Fetch all `circle_moments` rows for `circleId` ordered by `posted_at DESC`. Use existing `MomentService.fetchTodayMoments` approach but for all time (no date filter): `.eq("circle_id", value: circleId.uuidString)`.
3. Collect unique `userId` values from both result sets. Fetch display names from `profiles` table: `.from("profiles").select("id, display_name").in("id", values: userIds)`. Build a `[UUID: String]` lookup dict. If `profiles` table does not exist or returns no rows, fall back to `userId.uuidString.prefix(8)` as display name.
4. Map `ActivityFeedRow` → `FeedItem.habitCheckin(...)` or `FeedItem.streakMilestone(...)` based on `eventType`.
5. Map `CircleMomentRow` → `FeedItem.moment(...)`.
6. Merge the two arrays, sort by `sortTimestamp` descending (lexicographic ISO8601 sort is valid for UTC timestamps).
7. Slice: `let start = page * pageSize; let end = min(start + pageSize, all.count)`. Return `Array(all[start..<end])`. Return `[]` if `start >= all.count`.

**Implementation notes for `toggleReaction`:**

1. Check if a reaction from `userId` already exists on `itemId`: `.from("habit_reactions").select().eq("item_id", value: itemId.uuidString).eq("user_id", value: userId.uuidString).execute()`.
2. If existing reaction has the same emoji → delete it (user is un-reacting): `.from("habit_reactions").delete().eq("id", value: existing.id.uuidString).execute()`.
3. If existing reaction has different emoji → delete old, then insert new.
4. If no existing reaction → insert new row with `item_id`, `item_type`, `user_id`, `emoji`.

**Implementation notes for `fetchReactions`:**

Use `.in("item_id", values: itemIds.map { $0.uuidString })` to batch fetch.
