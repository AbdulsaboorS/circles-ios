Create `Circles/Services/CircleService.swift` following the exact SupabaseService.shared.client pattern:

```swift
import Foundation
import Supabase

@Observable
@MainActor
final class CircleService {
    static let shared = CircleService()
    private let client = SupabaseService.shared.client

    private init() {}

    /// Fetch all circles the current user belongs to via halaqa_members join
    func fetchMyCircles(userId: UUID) async throws -> [Circle] {
        try await client
            .from("halaqas")
            .select("*, halaqa_members!inner(user_id)")
            .eq("halaqa_members.user_id", value: userId.uuidString)
            .execute().value
    }

    /// Create a new circle. Caller becomes admin via halaqa_members insert.
    func createCircle(name: String, description: String?, prayerTime: String?, userId: UUID) async throws -> Circle {
        let inviteCode = generateInviteCode()
        let circle: Circle = try await client
            .from("halaqas")
            .insert([
                "name": name,
                "description": description ?? "",
                "created_by": userId.uuidString,
                "prayer_time": prayerTime ?? "",
                "invite_code": inviteCode
            ])
            .select()
            .single()
            .execute().value

        // Add creator as admin member
        try await client
            .from("halaqa_members")
            .insert([
                "halaqa_id": circle.id.uuidString,
                "user_id": userId.uuidString,
                "role": "admin"
            ])
            .execute()

        return circle
    }

    /// Look up circle by invite code, then join as member
    func joinByInviteCode(_ code: String, userId: UUID) async throws -> Circle {
        let circle: Circle = try await client
            .from("halaqas")
            .select()
            .eq("invite_code", value: code)
            .single()
            .execute().value

        try await client
            .from("halaqa_members")
            .insert([
                "halaqa_id": circle.id.uuidString,
                "user_id": userId.uuidString,
                "role": "member"
            ])
            .execute()

        return circle
    }

    /// Fetch all members of a circle
    func fetchMembers(circleId: UUID) async throws -> [HalaqaMember] {
        try await client
            .from("halaqa_members")
            .select()
            .eq("halaqa_id", value: circleId.uuidString)
            .execute().value
    }

    /// Generate random 8-character alphanumeric invite code
    private func generateInviteCode() -> String {
        let chars = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"
        return String((0..<8).map { _ in chars.randomElement()! })
    }
}
```

Key implementation notes:
- `@Observable @MainActor` matches AuthManager pattern from Phase 1
- `static let shared` singleton pattern matches SupabaseService and HabitService
- `fetchMyCircles` uses `halaqa_members!inner` join to only return circles user belongs to
- `createCircle` auto-generates 8-char invite code (no ambiguous chars: no 0/O/1/I/L)
- `joinByInviteCode` looks up circle first, then inserts membership row
- All methods are `async throws` matching Supabase SDK patterns
